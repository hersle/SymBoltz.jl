using SymBoltz: t, k, ϵ, D, O # use variables, derivative and perturbation operator
g = M.g # get metric of original model
pars = @parameters w₀ wₐ cₛ² Ω₀ # create parameters
vars = @variables ρ(t) P(t) w(t) δ(t) θ(t) σ(t) # create variables
eqs = [ # specify equations
    O(1)(w ~ w₀ + wₐ * (1 - g.a)) # equation of state
    O(1)(ρ ~ 3/(8*Num(π))*Ω₀ * g.a^(-3*(1+w₀+wₐ))) # energy density
    O(1)(P ~ w * ρ) # pressure
    O(ϵ)(D(δ) ~ -(1+w)*(θ-3*g.Φ) - 3*g.ℰ*(cₛ²-w)*δ) # overdensity
    O(ϵ)(D(θ) ~ -g.ℰ*(1-3*w)*θ - D(w)/(1+w)*θ + k^2*(cₛ²/(1+w)*δ + g.Ψ)) # momentum
    O(ϵ)(σ ~ 0) # shear stress
]
initialization_eqs = [ # initial conditions
    O(ϵ)(δ ~ -3/2 * (1+w) * g.Ψ)
    O(ϵ)(θ ~ 1/2 * (k^2/g.ℰ) * g.Ψ)
]
description = "w₀wₐ (CPL) dynamical dark energy"
@named X = ODESystem(eqs, t, vars, pars; initialization_eqs, description)

M = ΛCDM(Λ = X; name = "w₀wₐCDM") # recreate ΛCDM model, but replace CC by w0wa
params = merge(params, Dict(M.X.w₀ => -0.9, M.X.wₐ => 0.2)) # assign new parameters
prob = CosmologyProblem(M, params)
sol = solve(prob, ks)

\block[]{New features}{
    \begin{itemize}
    \item \textbf{Symbolic-numeric architecture:} Represents equations symbolically, and generates fast numerical code from them. Combines symbolic equation modeling and logic involving their structure (at compile time) with efficient numerical computing (at run time) into one integrated framework.
    \item \textbf{Modular physical structure:} Primarily structured around physical components (e.g. gravity and particle species). Easily replace one component with another without modifying the entire model. Build small and reusable physical components with everything defined in one place, and join them into full models. Accomodates extended models in a way that scales in model space. Computational stages (background, perturbations, \ldots) internally detected (semi-)automatically from the equations.
    \item \textbf{Approximation-free:} Always integrates full (stiff) equations with a wide selection of modern (implicit) solvers. No tight coupling, ultra-relativistic, radiation streaming or other approximations. Makes code simpler and extended models easier to implement.
    \item \textbf{Convenient:} Automatically calculate any variables or function thereof that are defined by some equations, even if not saved directly by the ODE solver. Eliminate chores one must do when extending traditional codes. Let user focus more on modeling and less on computing.
    \item \textbf{Differentiable:} Compute $\partial(\mathrm{output})/\partial(\mathrm{input})$ for any input and output with automatic differentiation (a method that calculates exact derivatives of a function using the chain rule on its basic operations). Important for gradient-based numerical methods, like efficient many-parameter inference methods.
    \end{itemize}
}

\block[bodyverticalshift=-1.75cm]{Goals}{
    \begin{itemize}
        \item \textbf{Create an integrated symbolic-numeric linear Einstein-Boltzmann modeling \emph{environment}} (not ``just'' a solver). Grow into a full linear + non-linear framework with $N$-body simulations?
        \item \textbf{Make it easier for modelers to build modified models} by avoiding complicating approximations and exploiting mixed symbolic-numeric computing that automates chores and provides helpful utilities.
                The dream is to \textbf{let the modeler simply write down \emph{one single set} of Einstein-Boltzmann equations for their cosmological model}.
                SymBoltz should \emph{automatically} detect the structure in the equations, split it up into stages (background $\rightarrow$ thermodynamics $\rightarrow$ perturbations $\rightarrow$ \ldots), spline functions between each stage, generate suitable initial conditions, change to a suitable independent variable, generate code and solve the model, compute or plot derived quantities, and so on.
        \item \textbf{Design a structure that is modular in terms of \emph{physics}} and scales better in model space.
                Rethink the traditional layout and avoid duplicating large monolithic codes for every combination of models into forks that are incompatible with each other.
        \item \textbf{Enable powerful gradient-based optimization/MCMC methods} (e.g. HMC/NUTS) by being fast and differentiable.
                Important for many-parameter inference from next-generation surveys!
    \end{itemize}
}
